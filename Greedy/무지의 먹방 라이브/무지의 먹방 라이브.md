from collections import deque

def solution(food_times, k):
    if sum(food_times) <= k:
        return -1
    d = []
    n = len(food_times)
    for i in range(1, n + 1):
        d.append([i, food_times[i - 1]])
    dq = deque(sorted(d, key=lambda x: x[1]))
    answer=0
    total=0
    while (1):
        ans = list(dq)
        ans = sorted(ans,key=lambda x: x[0])
        if dq[0][1]==0:
            dq.popleft()
            continue
        if k==0:
            answer = ans[0][0]
            break
        if k >= len(dq):
            m =min(dq,key= lambda x:x[1])[1]
            if m*len(dq) <=k:
                for i in dq:
                    i[1] -= m
                k -= m*len(dq)
            else:
                answer = dq[len(dq)%k+1][0]
        else:
            return ans[k][0]

    return answer
